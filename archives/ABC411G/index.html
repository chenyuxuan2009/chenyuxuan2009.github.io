题意

求 nnn 个点 mmm 条边的无向图中简单环数量。可以有重边，没有自环。

1≤n≤201\leq n\leq 201≤n≤20，1≤m≤2×1051\leq m\leq 2\times 10^51≤m≤2×105。

做法

有一个显然的 O(2nn3)\mathcal O(2^nn^3)O(2nn3) 的朴素状压 DP。

设 fs,i,maskf_{s,i,mask}fs,i,mask​ 表示开头为 sss 目前结尾为 iii 包含了 maskmaskmask 集合里的点，形成了一条链的方案数。转移枚举下一个点乘上两点之间边的个数即可。

最后把所有可能的开头结尾枚举一遍，乘上开头结尾间连边的数量，相加。

注意二元环一条边不能重复选，所以二元环的情况需要单独列出来跑一遍。

注意一个长度为 lll 的链这样子算会多算 2l2l2l 次，逆时针顺时针 222 种，起点 lll 种。

时间复杂度 O(2nn3)\mathcal O(2^nn^3)O(2nn3)，空间复杂度 O(2nn2)\mathcal O(2^nn^2)O(2nn2)，两者都爆了。具体见 https://atcoder.jp/contests/abc411/submissions/66940223。

优化空间考虑把枚举起点放在最外面。时间复杂度 O(2nn3)\mathcal O(2^nn^3)O(2nn3)，空间复杂度 O(2nn)\mathcal O(2^nn)O(2nn)，空间复杂度可以接受。具体见 https://atcoder.jp/contests/abc411/submissions/66941950。

我们考虑钦定 sss 作为起点之后，就把 sss 从图中删去，即我们只计算环中编号 min⁡\minmin 作为起点的情况。复杂度是类似等比数列的形式，即 O(∑i=1n2ii2)\mathcal O(\sum\limits_{i=1}^n 2^ii^2)O(i=1∑n​2ii2) 即 O(2nn2)\mathcal O(2^nn^2)O(2nn2)。

template <typename T>
void read(T &x)
{
    x = 0;
    int f = 1;
    char c = getchar();
    for (; c < '0' || c > '9'; c = getchar())
        if (c == '-')
            f = -f;
    for (; c >= '0' && c <= '9'; c = getchar())
        x = x * 10 + c - '0';
    x *= f;
}
template <typename T, typename... Args>

{
    read(x);
    read(y...);
}
template <class T>
void write(T x)
{
    static int stk[30];
    if (x < 0)
        putchar('-'), x = -x;
    int top = 0;
    do
    {
        stk[top++] = x % 10, x /= 10;
    } while (x);
    while (top)
        putchar(stk[--top] + '0');
}

template <int P>
class mod_int
{
    using Z = mod_int;

private:
    static int mo(int x) { return x < 0 ? x + P : x; }

public:
    int x;
    int val() const { return x; }
    mod_int() : x(0) {}
    template <class T>
    mod_int(const T &x_) : x(x_ >= 0 && x_ < P ? static_cast<int>(x_) : mo(static_cast<int>(x_ % P))) {}
    bool operator==(const Z &rhs) const { return x == rhs.x; }
    bool operator!=(const Z &rhs) const { return x != rhs.x; }
    Z operator-() const { return Z(x ? P - x : 0); }
    Z pow(long long k) const
    {
        Z res = 1, t = *this;
        while (k)
        {
            if (k & 1)
                res *= t;
            if (k >>= 1)
                t *= t;
        }
        return res;
    }
    Z &operator++()
    {
        x < P - 1 ? ++x : x = 0;
        return *this;
    }
    Z &operator--()
    {
        x ? --x : x = P - 1;
        return *this;
    }
    Z operator++(int)
    {
        Z ret = x;
        x < P - 1 ? ++x : x = 0;
        return ret;
    }
    Z operator--(int)
    {
        Z ret = x;
        x ? --x : x = P - 1;
        return ret;
    }
    Z inv() const { return pow(P - 2); }
    Z &operator+=(const Z &rhs)
    {
        (x += rhs.x) >= P && (x -= P);
        return *this;
    }
    Z &operator-=(const Z &rhs)
    {
        (x -= rhs.x) < 0 && (x += P);
        return *this;
    }
    Z operator-() { return -x; }
    Z &operator*=(const Z &rhs)
    {
        x = 1ULL * x * rhs.x % P;
        return *this;
    }
    Z &operator/=(const Z &rhs) { return *this *= rhs.inv(); }
#define setO(T, o)                                  \

const int P = 998244353;
using Z = mod_int<P>;
int n, m;
Z d[20][20];
Z f[20][1 << 20];
int main()
{
    read(n, m);
    while (m--)
    {
        int u, v;
        read(u, v);
        u--, v--;
        d[u][v]++, d[v][u]++;
    }
    Z ans = 0;
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < n; j++)
        {
            if (i == j)
                continue;
            ans += d[i][j] * (d[j][i] - 1) / 2;
        }
    }
    for (int s = 0; s < n; s++)
    {
        memset(f, 0, sizeof(f));
        f[s][1 << s] = 1;
        for (int S = 0; S < 1 << n; S++)
        {
            for (int i = s; i < n; i++)
            {
                if (f[i][S] == 0)
                    continue;
                if (!(S >> i & 1))
                    continue;
                for (int j = s + 1; j < n; j++)
                {
                    if (S >> j & 1)
                        continue;
                    f[j][S | (1 << j)] += f[i][S] * d[i][j];
                }
            }
        }
        for (int S = 0; S < 1 << n; S++)
        {
            if (__builtin_popcount(S) <= 2)
                continue;
            for (int i = s + 1; i < n; i++)
                ans += f[i][S] * d[i][s];
        }
    }
    cout << ans / 2 << '\n';
    return 0;
}
