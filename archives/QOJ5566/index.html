题意

给定 nnn 个点的树，点有点权，定义 f(u,v,op⁡)f(u,v,\operatorname{op})f(u,v,op) 表示 (ap0=uop⁡ap1op⁡ap2op⁡…op⁡apk=v)2(a_{p_0=u}\operatorname{op} a_{p_1}\operatorname{op} a_{p_2}\operatorname{op} \dots \operatorname{op} a_{p_k=v})^2(ap0​=u​opap1​​opap2​​op…opapk​=v​)2。即 u→vu\to vu→v 的路径上的点权 op⁡\operatorname{op}op 和的平方。

求出 ∑u=1n∑v=1nf(u,v,AND⁡)\sum\limits_{u=1}^n\sum\limits_{v=1}^n f(u,v,\operatorname{AND})u=1∑n​v=1∑n​f(u,v,AND)，∑u=1n∑v=1nf(u,v,OR⁡)\sum\limits_{u=1}^n\sum\limits_{v=1}^n f(u,v,\operatorname{OR})u=1∑n​v=1∑n​f(u,v,OR)，∑u=1n∑v=1nf(u,v,XOR⁡)\sum\limits_{u=1}^n\sum\limits_{v=1}^n f(u,v,\operatorname{XOR})u=1∑n​v=1∑n​f(u,v,XOR)。

答案对 998244353998244353998244353 取模。

人话版本：求树上两点路径点权按位与平方和，按位或平方和，按位异或平方和。

1≤n≤1051\leq n\leq 10^51≤n≤105，1≤ai≤2251\leq a_i\leq 2^{25}1≤ai​≤225。

原题时限 555 秒，模拟赛开到了 333 秒，我在 QOJ 卡到了 222 秒以内。

做法

糖糖题，难点在怎么卡常卡过去。微笑。

考虑拆位，我们每次考虑两个位 i,ji,ji,j。如果他们位上都是 111 里那么贡献是 2i+j2^{i+j}2i+j。展开一下平方和的构成不难看出。

直接大力 DP。设 fu,p,qf_{u,p,q}fu,p,q​ 表示目前 DP 到了点 uuu，从 uuu 子树内往上爬到 uuu，经过运算第 iii 位为 ppp，第 jjj 位为 qqq，共有 fu,p,qf_{u,p,q}fu,p,q​ 个方案。

合并的时候在 LCA 也就是 uuu 处合并 uuu 的两个儿子 v1,v2v_1,v_2v1​,v2​。把经过运算能得到 p=q=1p=q=1p=q=1 的状态乘起来。

然后把三个运算的 DP 都写一遍。恭喜你通过了原题。

我这么写了一下本机跑了 101010 秒，稍微卡了一下 QOJ 跑了 444 秒出头，过了但是模拟赛肯定过不去，所以我们要进行精细卡常。微笑。

DFS 可以换成倒序枚举 DFS 序，这样会非常非常快。此事在 CF2035F https://codeforces.com/contest/2035/submission/288369192 被卡常压哨过题 https://codeforces.com/contest/2035/submission/288370122 亦有记载。

拆位的时候我们只需要枚举 i≤ji\leq ji≤j 的情况，i>ji\gt ji>j 是同理的。如果 i=ji=ji=j 那么贡献少个 ×2\times 2×2 就好。转移的时候特判一下。

三个操作可以都用一个 namespace 封装起来。

发现 AND⁡\operatorname{AND}AND 操作只需要考察 fu,1,1f_{u,1,1}fu,1,1​ 的值。那么我们在对应的 namespace 里开 fu′f'_ufu′​ 表示原来的 fu,1,1f_{u,1,1}fu,1,1​ 即可。而且 bit⁡(au,i)=0\operatorname{bit}(a_u,i)=0bit(au​,i)=0 或 bit⁡(au,j)=0\operatorname{bit}(a_u,j)=0bit(au​,j)=0 可以直接忽略这个点 uuu 的转移。

OR⁡\operatorname{OR}OR 操作转移太复杂了，我们考虑容斥一下。把所有位都看做 111 然后减掉一些不存在的情况。我们在预处理 DFS 序的时候就可以把都看做 111 的答案处理出来。剩下的两个发现转移都需要满足此时 bit⁡(au,i)=0\operatorname{bit}(a_u,i)=0bit(au​,i)=0 或 bit⁡(au,j)=0\operatorname{bit}(a_u,j)=0bit(au​,j)=0 才有转移，剪枝一下。

XOR⁡\operatorname{XOR}XOR 这个不是很好卡。至少我不太会。

然后来点经典操作，循环展开，把每个循环对应的转移全写下来。然后合并一下相同的乘法，例如 OR⁡\operatorname{OR}OR 的转移有 tmp -= 1LL * f[u][0][0] * (f[v][0][0] + f[v][1][0]);。写两个乘法会变慢，不太懂，反正我把这个优化掉了。

循环展开之后这个是可以并行操作什么的？反正更快了。

发现 fff 数组用 int 可以存的下。那么我们存成 int，转移的时候加上 1LL 强转一下类型，然后我们只需要给答案开成 long long 类型。这样 memset 清空数组会更快。事实上开成无符号更快，波波说是因为无符号的 uint 和 ull 转换没有代价。

每次计算 2i+j2^{i+j}2i+j 太慢了，发现答案每次都要带一个这个，我们放在最后乘上去。

取模可以使用 Barrett 模乘。这个其实不是很重要。push_back 换成 emplace_back。这个也不是很重要。第一次预处理的时候可以在邻接表内删掉父亲防止 DFS 序要多记一个父亲。

这些都卡掉过这题绰绰有余了。现在这份代码是 QOJ 最优解第一页（硬气

至此我在 QOJ 不开火车头的情况下卡到了 1666ms1666\text{ms}1666ms。然后模拟赛忘开 freopen 了哈哈。太搞笑了。

不太懂卡常理论，但是感谢波波的指导，不知道会不会被 O2 优化。

#include <bits/stdc++.h>
using namespace std;
#define getchar() p1 == p2 && (p2 = (p1 = buf) + fread(buf, 1, 1000000, stdin), p1 == p2) ? EOF : *p1++
char buf[1000000], *p1 = buf, *p2 = buf;
template <typename T>
void read(T &x)
{
    x = 0;
    int f = 1;
    char c = getchar();
    for (; c < '0' || c > '9'; c = getchar())
        if (c == '-')
            f = -f;
    for (; c >= '0' && c <= '9'; c = getchar())
        x = x * 10 + c - '0';
    x *= f;
}
template <typename T, typename... Args>
void read(T &x, Args &...y)
{
    read(x);
    read(y...);
}
template <class T>
void write(T x)
{
    static int stk[30];
    if (x < 0)
        putchar('-'), x = -x;
    int top = 0;
    do
    {
        stk[top++] = x % 10, x /= 10;
    } while (x);
    while (top)
        putchar(stk[--top] + '0');
}
template <class T>
void write(T x, char lastChar) { write(x), putchar(lastChar); }
#define FIO(s)                    \
    freopen(s ".in", "r", stdin); \
    freopen(s ".out", "w", stdout);
typedef long long ll;
const int p = 998244353;
typedef __int128 LL;
struct Mod
{
    const ll m = ((LL)1 << 64) / p;
    inline ll operator()(ll x) { return x - ((LL(x) * m) >> 64) * p; }
} mod;
int n;
ll a[100020];
vector<int> g[100020];
ll ans;
ll s, t;
vector<int> vec;
int dfs(int u, int p)
{
    int s = 1;
    for (int v : g[u])
    {
        if (v == p)
            continue;
        int tmp = dfs(v, u);
        t += 1LL * s * tmp;
        s += tmp;
    }
    g[u].erase(remove(g[u].begin(), g[u].end(), p), g[u].end());
    vec.emplace_back(u);
    return s;
}
namespace AND
{
    int f[100020];
    void clear()
    {
        memset(f, 0, sizeof(f));
    }
    void calc(int i, int j)
    {
        ll tmp = 0;
        for (int u : vec)
        {
            int p = a[u] >> i & 1;
            int q = a[u] >> j & 1;
            if (!p || !q)
                continue;
            f[u] = 1;
            for (int v : g[u])
            {
                tmp += 1LL * f[u] * f[v];
                f[u] += f[v];
            }
        }
        ans += mod(1LL << i + j + 1 + (i != j)) * mod(tmp);
        ans = mod(ans);
    }
    void calc()
    {
        ans = 0;
        for (int i = 0; i < 25; i++)
        {
            for (int j = i; j < 25; j++)
            {
                clear();
                calc(i, j);
            }
        }
    }
}
namespace OR
{
    int f[100020][2][2];
    void clear()
    {
        memset(f, 0, sizeof(f));
    }
    void calc(int i, int j)
    {
        ll tmp = t;
        for (int u : vec)
        {
            int p = a[u] >> i & 1;
            int q = a[u] >> j & 1;
            if (p && q)
                continue;
            f[u][p][q] = 1;
            for (int v : g[u])
            {
                if (!q)
                {
                    tmp -= 1LL * f[u][0][0] * (f[v][0][0] + f[v][1][0]);
                    tmp -= 1LL * f[u][1][0] * (f[v][0][0] + f[v][1][0]);
                }
                if (!p)
                {
                    tmp -= 1LL * f[u][0][0] * (f[v][0][0] + f[v][0][1]);
                    tmp -= 1LL * f[u][0][1] * (f[v][0][0] + f[v][0][1]);
                }
                tmp += 1LL * f[u][0][0] * f[v][0][0];
                f[u][p][q] += f[v][0][0];
                f[u][p][1] += f[v][0][1];
                f[u][1][q] += f[v][1][0];
                f[u][1][1] += f[v][1][1];
            }
        }
        ans += mod(1LL << i + j + 1 + (i != j)) * mod(tmp);
        ans = mod(ans);
    }
    void calc()
    {
        ans = 0;
        for (int i = 0; i < 25; i++)
        {
            for (int j = i; j < 25; j++)
            {
                clear();
                calc(i, j);
            }
        }
    }
}
namespace XOR
{
    int f[100020][2][2];
    void clear()
    {
        memset(f, 0, sizeof(f));
    }
    void calc(int i, int j)
    {
        ll tmp = 0;
        for (int u : vec)
        {
            int p = a[u] >> i & 1;
            int q = a[u] >> j & 1;
            f[u][p][q] = 1;
            for (int v : g[u])
            {
                tmp += 1LL * f[u][0][0] * f[v][1][1];
                tmp += 1LL * f[u][0][1] * f[v][1][0];
                tmp += 1LL * f[u][1][0] * f[v][0][1];
                tmp += 1LL * f[u][1][1] * f[v][0][0];
                f[u][p][q] += f[v][0][0];
                f[u][p][q ^ 1] += f[v][0][1];
                f[u][p ^ 1][q] += f[v][1][0];
                f[u][p ^ 1][q ^ 1] += f[v][1][1];
            }
        }
        ans += mod(1LL << i + j + 1 + (i != j)) * mod(tmp);
        ans = mod(ans);
    }
    void calc()
    {
        ans = 0;
        for (int i = 0; i < 25; i++)
        {
            for (int j = i; j < 25; j++)
            {
                clear();
                calc(i, j);
            }
        }
    }
}
int main()
{
    // FIO("tree");
    read(n);
    for (int i = 1; i <= n; i++)
        read(a[i]);
    for (int i = 1; i <= n; i++)
        s += a[i] * a[i], s = mod(s);
    for (int i = 1; i < n; i++)
    {
        int u, v;
        read(u, v);
        g[u].emplace_back(v);
        g[v].emplace_back(u);
    }
    dfs(1, 0);
    t = mod(t);
    AND::calc();
    write((mod(ans + s) % p + p) % p, ' ');
    OR::calc();
    write((mod(ans + s) % p + p) % p, ' ');
    XOR::calc();
    write((mod(ans + s) % p + p) % p, ' ');
    putchar('\n');
    // cerr << fixed << setprecision(10) << 1.0 * clock() / CLOCKS_PER_SEC << '\n';
    return 0;
}
