abc257_a

题目翻译

给定一个 nnn，生成一个字符串，内容为 nnn 个 A\tt{A}A，nnn 个 B\tt{B}B，一直到 nnn 个 Z\tt{Z}Z。

输出第 kkk 位。

题目思路

直接模拟到第 nnn 位。

这是最暴力的写法。

也可以用数学知识算一下当前是哪个字符。

我采用了最暴力的。

AC 代码

int main()
{
    int n, x, cnt = 0;
    cin >> n >> x;
    for (int i = 1; i <= 26; i++)
    {
        for (int j = 1; j <= n; j++)
        {
            cnt++;
            if (cnt == x)
            {
                cout << char (i + 'A' - 1) << endl;
                return 0;
            }
        }
    }
    return 0;
}

abc257_c

题目翻译

有 nnn 个人，每个人有个体重 wiw_iwi​，以及 0\tt{0}0 代表他是小孩，1\tt{1}1 代表成人。

我们认为，比 XXX 轻的是小孩，反之是成人。

请选择一个 XXX，使身份匹配正确的人数最多。

题目思路

把这些人按体重排序，相同体重成人在前。

统计这里一共多少成人。

对于每个 [1,i][1,i][1,i] 的区间，算一下类似前缀和的操作。

是小孩，之前统计的加 111，反之 −1-1−1。

AC 代码

struct node
{
    bool f;
    int w;
} a[200005];
bool cmp (node x, node y)
{
    if (x.w == y.w)
    {
        return x.f > y.f;
    }
    return x.w < y.w;
}
int main()
{
    int n, ch = 0, ans;
    cin >> n;
    char c;
    for (int i = 1; i <= n; i++)
    {
        cin >> c;
        if (c == '0')
        {
            a[i].f = 0;
        }
        else
        {
            a[i].f = 1;
            ch++;
        }
    }
    for (int i = 1; i <= n; i++)
    {
        cin >> a[i].w;
    }
    sort (a + 1, a + n + 1, cmp);
    ans = ch;
    for (int i = 1; i <= n; i++)
    {
        if (a[i].f == 0)
        {
            ch++;
        }
        else
        {
            ch--;
        }
        ans = max (ans, ch);
    }
    cout << ans << endl;
    return 0;
}

abc257_d

题目翻译

有 nnn 个蹦床，每个蹦床坐标为 (xi,yi)(x_i,y_i)(xi​,yi​)，并拥有主场 buff pip_ipi​ 加持。

（这样说是不是通俗易懂

可以自选起点。

如果从第 iii 个往第 jjj 个跳，所需 S×pi≥∣xi−xj∣+∣yi−yj∣S\times p_i\geq \left|x_i-x_j\right|+\left|y_i-y_j\right|S×pi​≥∣xi​−xj​∣+∣yi​−yj​∣ 的必备条件。

问 SSS 最小多少。

题目思路

Floyd 最短路。

先求出从 iii 跳到 jjj 的所需 SSS，再用 Floyd 做最短路，看经过中间过渡，能否缩小 SSS。

转移即为 fi,j=min⁡(max⁡(fi,k,fk,j),fi,j)f_{i,j} = \min(\max (f_{i,k}, f_{k,j}), f{i,j})fi,j​=min(max(fi,k​,fk,j​),fi,j)。

AC 代码

ll x[205], y[205], p[205];
ll f[205][205];
int main()
{
    ll n;
    cin >> n;
    for (int i = 1; i <= n; i++)
    {
        cin >> x[i] >> y[i] >> p[i];
    }
    for (int i = 1; i <= n; i++)
    {
        for (int j = 1; j <= n; j++)
        {
            ll s = abs (x[i] - x[j]) + abs (y[i] - y[j]);
            if(s%p[i]==0)f[i][j]=s/p[i];
            else f[i][j]=s/p[i]+1;
        }
    }
    for (int k = 1; k <= n; k++)
    {
        for (int i = 1; i <= n; i++)
        {
            for (int j = 1; j <= n; j++)
            {
                f[i][j] = min (max (f[i][k], f[k][j]), f[i][j]);
            }
        }
    }
    ll ans = 1e20;
    for (int i = 1; i <= n; i++)
    {
        ll x = -1e20;
        for (int j = 1; j <= n; j++)
        {
            x = max (f[i][j], x);
        }
        ans = min (ans, x);
    }
    cout << ans << endl;
    return 0;
}

abc257_e

题目翻译

有 999 种数字，1,2,…91,2,\dots91,2,…9。

数字 iii 可以花 cic_ici​ 的价格购买，你有 nnn 元。

最后输出你购买的数字拼成的最大值。

题目思路

我们考虑，先让位数最大。

然后，再最大化最高位、次高位、次次高位，一直往下。

最大位数，即为用 nnn 除以最便宜的价钱。

然后，你可能会有剩余的钱。

剩余的，我们开始置换。

从 999 开始枚举，能换就换，并输出你置换的数字。

这样最大化了较高位，还保证位数没有变少。

具体可以品味下代码。

AC 代码

int c[20];
int main()
{
    int n, mnc = INT_MAX, maxn;
    cin >> n;
    for (int i = 1; i <= 9; i++)
    {
        cin >> c[i];
        if (c[i] <= mnc)
        {
            mnc = c[i];
            maxn = i;
        }
    }
    int cnt = n / mnc;
    int x = 0, cnt2 = 0;
    int s = n - cnt * mnc;
    for (int i = 9; i >= 1; i--)
    {
        if (i > maxn)
        {
            if (c[i] - mnc <= s)
            {
                cnt2 = s / (c[i] - mnc);
                cnt -= cnt2;
                s%=c[i] - mnc;
                for (int j = 1; j <= cnt2; j++)
                {
                    cout << i;
                }
            }
        }
    }
    for (int i = 1; i <= cnt; i++)
    {
        cout << maxn;
    }
    return 0;
}
