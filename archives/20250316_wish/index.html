题意

给定长度为 nnn 的数列 aaa 和质数 ppp，有 qqq 次操作：

  1. 给定 l,x,rl,x,rl,x,r，区间乘法，ai←ai×x,∀i∈[l,r]a_i\gets a_i\times x,\forall i\in[l,r]ai​←ai​×x,∀i∈[l,r]。
  2. 给定 l,rl,rl,r，求 {ai∣i∈[l,r]}\{a_i\mid i\in[l,r]\}{ai​∣i∈[l,r]} 这个集合在  mod p\bmod pmodp 乘法群下的的生成子群的阶。

1≤n,q≤1051\leq n,q\leq 10^51≤n,q≤105，1≤p≤1091\leq p\leq 10^91≤p≤109。可以做到 1≤p≤10141\leq p\leq 10^{14}1≤p≤1014，梦熊加强到了这个版本。

https://mna.wang/contest/1932/problem/1。

https://qoj.ac/problem/8426

做法

考虑取离散对数。之后生成子群就是加法群了。就是求线性组合能组合出多少数，考虑裴蜀定理，那么就是能组成 p−1gcd⁡(al,al+1,…,ar,p−1)\frac{p-1}{\gcd(a_l,a_{l+1},\dots,a_r,p-1)}gcd(al​,al+1​,…,ar​,p−1)p−1​ 个数。

转化成区间加法区间 gcd⁡\gcdgcd。考虑 gcd⁡(x,y)=gcd⁡(x,y−x)\gcd(x,y)=\gcd(x,y-x)gcd(x,y)=gcd(x,y−x)，所以对序列差分之后单点修改区间求 gcd⁡\gcdgcd。

瓶颈在求离散对数，需要 BSGS。根号复杂度，能过原题，过不了加强。

考虑到 xxx 的阶 δ(x)\delta(x)δ(x) 就是 {x}\{x\}{x} 的生成子群的大小。阶为 aaa 的群和阶为 bbb 的群分别为 ⟨g(p−1)/a⟩\langle g^{(p-1)/a} \rangle⟨g(p−1)/a⟩ 和 ⟨g(p−1)/b⟩\langle g^{(p-1)/b} \rangle⟨g(p−1)/b⟩，合并为 ⟨ggcd⁡((p−1)/a,(p−1)/b)⟩\langle g^{\gcd((p-1)/a,(p-1)/b)} \rangle⟨ggcd((p−1)/a,(p−1)/b)⟩。大小为 p−1gcd⁡((p−1)/a,(p−1)/b)=lcm⁡(a,b)\frac{p-1}{\gcd((p-1)/a,(p-1)/b)}=\operatorname{lcm}(a,b)gcd((p−1)/a,(p−1)/b)p−1​=lcm(a,b)。

所以区间乘法区间 lcm⁡\operatorname{lcm}lcm 即可。

依旧商分，然后单点修改区间 lcm⁡\operatorname{lcm}lcm。

求阶预处理质因子然后试除可以做到 O(log⁡2p)\mathcal O(\log^2 p)O(log2p)。总复杂度 O((n+q)log⁡2p)\mathcal O((n+q)\log^2 p)O((n+q)log2p)。

#include <bits/stdc++.h>
using namespace std;
#define getchar() p1 == p2 && (p2 = (p1 = buf) + fread(buf, 1, 1000000, stdin), p1 == p2) ? EOF : *p1++
char buf[1000000], *p1 = buf, *p2 = buf;
template <typename T>
void read(T &x)
{
    x = 0;
    int f = 1;
    char c = getchar();
    for (; c < '0' || c > '9'; c = getchar())
        if (c == '-')
            f = -f;
    for (; c >= '0' && c <= '9'; c = getchar())
        x = x * 10 + c - '0';
    x *= f;
}
template <typename T, typename... Args>
void read(T &x, Args &...y)
{
    read(x);
    read(y...);
}
template <class T>
void write(T x)
{
    static int stk[30];
    if (x < 0)
        putchar('-'), x = -x;
    int top = 0;
    do
    {
        stk[top++] = x % 10, x /= 10;
    } while (x);
    while (top)
        putchar(stk[--top] + '0');
}
template <class T>
void write(T x, char lastChar) { write(x), putchar(lastChar); }
#define FIO(s)                    \
    freopen(s ".in", "r", stdin); \
    freopen(s ".out", "w", stdout);
typedef long long ll;
typedef __int128 LL;
LL lcm(LL x, LL y) { return (LL)x / __gcd(x, y) * y; }
ll p;
LL qpow(LL x, LL y)
{
    LL ret = 1;
    for (; y > 0; y >>= 1)
    {
        if (y & 1)
            (ret *= x) %= p;
        (x *= x) %= p;
    }
    return ret;
}
int n, q;
LL a[100020];
vector<ll> fac;
void init(ll p)
{
    for (ll i = 2; i <= sqrtl(p); i++)
    {
        while (!(p % i))
        {
            fac.emplace_back(i);
            p /= i;
        }
    }
    if (p > 1)
        fac.emplace_back(p);
}
ll ord(ll x)
{
    ll o = p - 1;
    for (ll i : fac)
    {
        if (qpow(x, o / i) == 1)
            o /= i;
    }
    return o;
}
struct BIT
{
    int n;
    LL c[100020];
    void init(int m)
    {
        n = m;
        for (int i = 0; i <= n; i++)
            c[i] = 1;
    }
    int lowbit(int x) { return x & -x; }
    void mul(int x, LL y)
    {
        for (int i = x; i <= n; i += lowbit(i))
            (c[i] *= y) %= p;
    }
    LL query(int x)
    {
        LL ret = 1;
        for (int i = x; i > 0; i &= i - 1)
            (ret *= c[i]) %= p;
        return ret;
    }
} bit;
struct SegTree
{
    struct node
    {
        LL lcm;
    } t[100020 << 2];
#define ls id << 1
#define rs id << 1 | 1
    void push_up(int id) { t[id].lcm = lcm(t[ls].lcm, t[rs].lcm); }
    void build(int id = 1, int l = 1, int r = n)
    {
        if (l == r)
            return t[id].lcm = ord(bit.query(l) * qpow(bit.query(l - 1), p - 2) % p), void();
        int mid = l + r >> 1;
        build(ls, l, mid);
        build(rs, mid + 1, r);
        push_up(id);
    }
    void update(int pos, int id = 1, int l = 1, int r = n)
    {
        if (r < pos || l > pos)
            return;
        if (pos <= l && r <= pos)
            return t[id].lcm = ord(bit.query(l) * qpow(bit.query(l - 1), p - 2) % p), void();
        int mid = l + r >> 1;
        update(pos, ls, l, mid);
        update(pos, rs, mid + 1, r);
        push_up(id);
    }
    LL query(int ql, int qr, int id = 1, int l = 1, int r = n)
    {
        if (r < ql || l > qr)
            return 1;
        if (ql <= l && r <= qr)
            return t[id].lcm;
        int mid = l + r >> 1;
        return lcm(query(ql, qr, ls, l, mid), query(ql, qr, rs, mid + 1, r));
    }
} sgt;
int main()
{
    // FIO("ds");
    read(p, n, q);
    for (int i = 1; i <= n; i++)
        read(a[i]);
    init(p - 1);
    bit.init(n);
    a[0] = 1;
    for (int i = 1; i <= n; i++)
        bit.mul(i, a[i] * qpow(a[i - 1], p - 2) % p);
    sgt.build();
    while (q--)
    {
        int op, l, r;
        read(op, l, r);
        if (op & 1)
        {
            ll x;
            read(x);
            bit.mul(l, x);
            if (r < n)
                bit.mul(r + 1, qpow(x, p - 2));
            sgt.update(l);
            if (r < n)
                sgt.update(r + 1);
        }
        else
        {
            LL x = ord(bit.query(l));
            write(l == r ? x : lcm(x, sgt.query(l + 1, r)), '\n');
        }
    }
    return 0;
}
