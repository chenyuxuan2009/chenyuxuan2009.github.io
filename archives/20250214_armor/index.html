题意

给定 n,mn,mn,m，表示有两个人 A 和 B 都有一个大小为 nnn 的背包，mmm 个现限制，给出 x,y,a,bx,y,a,bx,y,a,b，表示这个物品要么装到 A 的第 xxx 个位置给 A 产生 aaa 的贡献，要么装到 B 的第 yyy 个位置给 B 产生 bbb 的贡献。保证存在一个合法的分配方案。设 XXX 为 A 的贡献和，YYY 为 B 的贡献和，求 min⁡XY\min XYminXY。

1≤n≤5×1051\leq n\leq 5\times 10^51≤n≤5×105，1≤m≤2n1\leq m\leq 2n1≤m≤2n，1≤x,y≤n1\leq x,y\leq n1≤x,y≤n，1≤a,b≤1061\leq a,b\leq 10^61≤a,b≤106，1≤∑a,∑b≤2×1091\leq \sum a,\sum b\leq 2\times 10^91≤∑a,∑b≤2×109。

题目做法

首先连出二分图，x↔y+nx \leftrightarrow y+nx↔y+n。

由于一定有解，所以连通块一定形成一棵树或者一棵基环树。

一棵树 TTT 的方案有 ∣T∣|T|∣T∣ 种。枚举每个点作为根，之后所有边都摊到儿子上。可以用换根 DP 求出所有的方案。

一棵基环树方案只有 222 种，给环定向之后可以确定其他所有点。

那么我们这样子搜出来之后，问题转化成了：

给定 nnn，对于每个 1≤i≤n1\leq i\leq n1≤i≤n 给定 mmm 个 (x,y)(x,y)(x,y)，对每个 iii 选一组 (x′,y′)(x',y')(x′,y′)，最小化 (∑x′)×(∑y′)(\sum x') \times (\sum y')(∑x′)×(∑y′)。

我们把每个 iii 的 mmm 个 (x,y)(x,y)(x,y) 看成平面上的点，那么如果存在 x′≤xx'\leq xx′≤x 且 y′≤xy'\leq xy′≤x 那么可以删去 (x,y)(x,y)(x,y)。也就是我们只需要维护下凸壳。

我们把这 nnn 个下凸壳合并起来就可以了。也就是对 nnn 个下凸壳做闵可夫斯基和。

复杂度带一个 log⁡\loglog。

namespace Geometry
{
    using T = ll;
    constexpr T eps = 0;

    bool eq(const T &x, const T &y) { return abs(x - y) <= eps; }
    inline constexpr int type(T x, T y)
    {
        if (x == 0 and y == 0)
            return 0;
        if (y < 0 or (y == 0 and x > 0))
            return -1;
        return 1;
    }
    struct Point
    {
        T x, y;
        constexpr Point(T _x = 0, T _y = 0) : x(_x), y(_y) {}
        constexpr Point operator+() const noexcept { return *this; }
        constexpr Point operator-() const noexcept { return Point(-x, -y); }
        constexpr Point operator+(const Point &p) const { return Point(x + p.x, y + p.y); }
        constexpr Point operator-(const Point &p) const { return Point(x - p.x, y - p.y); }
        constexpr Point &operator+=(const Point &p) { return x += p.x, y += p.y, *this; }
        constexpr Point &operator-=(const Point &p) { return x -= p.x, y -= p.y, *this; }
        constexpr T operator*(const Point &p) const { return x * p.x + y * p.y; }
        constexpr Point &operator*=(const T &k) { return x *= k, y *= k, *this; }
        constexpr Point operator*(const T &k) { return Point(x * k, y * k); }
        constexpr bool operator==(const Point &r) const noexcept { return r.x == x and r.y == y; }
        constexpr T cross(const Point &r) const { return x * r.y - y * r.x; }

        constexpr bool operator<(const Point &r) const { return pair(x, y) < pair(r.x, r.y); }

        // 1 : left, 0 : same, -1 : right
        constexpr int toleft(const Point &r) const
        {
            auto t = cross(r);
            return t > eps ? 1 : t < -eps ? -1
                                          : 0;
        }

        constexpr bool arg_cmp(const Point &r) const
        {
            int L = type(x, y), R = type(r.x, r.y);
            if (L != R)
                return L < R;

            T X = x * r.y, Y = r.x * y;
            if (X != Y)
                return X > Y;
            return x < r.x;
        }
    };
    bool arg_cmp(const Point &l, const Point &r) { return l.arg_cmp(r); }

    struct Line
    {
        Point a, b;
        Line() = default;

        Line(T A, T B, T C)

            }
            else
            {
                a = Point(0, C / B), b = Point(C / A, 0);
            }

    using pt = Point;
    using Points = vector<pt>;
    using Polygon = Points;
    T cross(const pt &x, const pt &y) { return x.x * y.y - x.y * y.x; }
    T dot(const pt &x, const pt &y) { return x.x * y.x + x.y * y.y; }

    T abs2(const pt &x) { return dot(x, x); }

    Points convex_hull(Points p)
    {
        int n = p.size(), k = 0;
        if (n <= 2)
            return p;
        sort(begin(p), end(p), [](pt x, pt y)
             { return (x.x != y.x ? x.x < y.x : x.y < y.y); });
        Points ch(2 * n);
        for (int i = 0; i < n; ch[k++] = p[i++])
        {
            while (k >= 2 && cross(ch[k - 1] - ch[k - 2], p[i] - ch[k - 1]) <= 0)
                --k;
        }
        for (int i = n - 2, t = k + 1; i >= 0; ch[k++] = p[i--])
        {
            while (k >= t && cross(ch[k - 1] - ch[k - 2], p[i] - ch[k - 1]) <= 0)
                --k;
        }
        ch.resize(k - 1);
        return ch;
    }

    Polygon Minkowski_sum(const vector<Polygon> Q)
    {
        vector<vector<Segment>> e(Q.size());
        int sum = 0;
        const auto cmp = [](const Segment &u, const Segment &v)
        { return (u.b - u.a).arg_cmp(v.b - v.a); };
        vector<Segment> ed;
        Point u = (0, 0);
        for (auto &P : Q)
        {
            vector<Segment> e1(P.size());
            for (int i = 0; i < P.size(); i++)
                e1[i] = {P[i], P[(i + 1) % P.size()]};
            rotate(begin(e1), min_element(e1.begin(), e1.end(), cmp), end(e1));
            u += e1[0].a;
            sum += e1.size();
            for (auto &i : e1)
                ed.push_back(i);
        }
        sort(ed.begin(), ed.end(), cmp);
        const auto check = [](const Points &res, const Point &u)
        {
            const auto back1 = res.back(), back2 = *prev(end(res), 2);
            return eq(cross(back1 - back2, u - back2), eps) and dot(back1 - back2, u - back1) >= -eps;
        };
        Points res{u};
        res.reserve(sum);
        for (const auto &v : ed)
        {
            u = u + v.b - v.a;
            while (res.size() >= 2 and check(res, u))
                res.pop_back();
            res.emplace_back(u);
        }
        if (res.size() and check(res, res[0]))
            res.pop_back();
        return res;
    }
} // namespace Geometry

        else
            Y += w2, X -= w1;

}
